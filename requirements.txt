import streamlit as st

# -------------------------------
# 행동별 취미 추천 데이터
# -------------------------------
hobbies = {
    "🌳 밖에서 활동하기": ["⛰️ 등산", "🚴 자전거 타기", "🏕️ 캠핑"],
    "🏠 집에서 조용히 있기": ["📖 독서", "🎨 그림 그리기", "✍️ 글쓰기"],
    "🎶 음악/예술 즐기기": ["🎤 노래 부르기", "🎹 피아노 연주", "📸 사진 찍기"],
    "🤝 사람들과 어울리기": ["🃏 보드게임", "🍽️ 요리 모임", "⚽ 축구"],
    "🧩 새로운 걸 배우기": ["🖥️ 코딩", "🌍 외국어 공부", "🎬 영화 감상"],
    "💪 몸을 움직이기": ["🏋️ 헬스", "🥊 복싱", "🕺 댄스"],
}

# -------------------------------
# 앱 기본 설정
# -------------------------------
st.set_page_config(page_title="취미 추천 사이트", page_icon="🎯", layout="centered")

st.title("✨ 취미 추천 사이트 ✨")
st.markdown("좋아하는 행동을 선택하면 어울리는 취미를 추천해드릴게요! 🏖️")

# 행동 선택
action = st.radio(
    "어떤 행동을 좋아하시나요?",
    list(hobbies.keys())
)

# 버튼 클릭 시 결과 출력
if st.button("🔮 내 취미 추천 받기"):
    st.subheader(f"🌟 '{action}' 좋아하는 당신에게 추천하는 취미 🌟")
    for hobby in hobbies[action]:
        st.write(f"- {hobby}")

    st.success("🔥 새로운 취미로 삶에 활력을 불어넣어보세요! 🚀")

# -------------------------------
# 푸터
# -------------------------------
st.markdown("---")
st.caption("Made with ❤️ using Streamlit | Hobby Recommender")
